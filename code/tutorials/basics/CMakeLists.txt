cmake_minimum_required(VERSION 2.6)

BFG_INIT_PACKAGE(TutorialBasics)

FILE(GLOB SOURCES
	*.cpp
)

FILE(GLOB HEADERS
	*.h
)

INCLUDE_DIRECTORIES(${BoostLog_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${Boost_INCLUDE_DIRS})
INCLUDE_DIRECTORIES(${OGRE_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${OIS_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${MYGUI_INCLUDE_DIR})
INCLUDE_DIRECTORIES(${BFG_ENGINE_INCLUDE_DIR})
INCLUDE_DIRECTORIES(..)	# For subdirectories
INCLUDE_DIRECTORIES(.)	# For subdirectories

ADD_EXECUTABLE(bfgTutorialBasics ${SOURCES} ${HEADERS})

SET_TARGET_PROPERTIES(bfgTutorialBasics PROPERTIES OUTPUT_NAME_DEBUG "bfgTutorialBasics_d")

TARGET_LINK_LIBRARIES(bfgTutorialBasics
	${Boost_LIBRARIES}
	${BoostLog_LIBRARIES}
	${OGRE_LIBRARIES}
	${MYGUI_LIBRARIES}
	${MYGUI_OGRE_LIBRARIES}
	bfgBase
	bfgController
	bfgCore
	bfgEventCore
	bfgModel
	bfgView
)

INSTALL(TARGETS bfgTutorialBasics
	RUNTIME DESTINATION ${BFG_BIN_DIR}
	COMPONENT devel
)
